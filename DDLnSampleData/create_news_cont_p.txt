DROP TABLE PNWSOWN.NEWS_CONT_P CASCADE CONSTRAINTS;

CREATE TABLE PNWSOWN.NEWS_CONT_P
(
  YMD       VARCHAR2(8 BYTE)                    NOT NULL,
  SEQNO     VARCHAR2(8 BYTE)                    NOT NULL,
  NEWSCODE  VARCHAR2(16 BYTE)                   NOT NULL,
  LINENO    NUMBER(9)                           NOT NULL,
  CONTENT   VARCHAR2(132 BYTE)
)
TABLESPACE TS_DAT_NWS01_08
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          1M
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE;

COMMENT ON TABLE PNWSOWN.NEWS_CONT_P IS '뉴스핌';

COMMENT ON COLUMN PNWSOWN.NEWS_CONT_P.YMD IS '일자';

COMMENT ON COLUMN PNWSOWN.NEWS_CONT_P.SEQNO IS '순번';

COMMENT ON COLUMN PNWSOWN.NEWS_CONT_P.NEWSCODE IS '뉴스코드';

COMMENT ON COLUMN PNWSOWN.NEWS_CONT_P.LINENO IS '라인번호';

COMMENT ON COLUMN PNWSOWN.NEWS_CONT_P.CONTENT IS '내용';


CREATE UNIQUE INDEX PNWSOWN.NEWS_CONT_P_PK ON PNWSOWN.NEWS_CONT_P
(YMD, SEQNO, NEWSCODE, LINENO)
LOGGING
TABLESPACE TS_IDX_NWS01_08
PCTFREE    10
INITRANS   2
MAXTRANS   255
STORAGE    (
            INITIAL          1M
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           );

ALTER TABLE PNWSOWN.NEWS_CONT_P ADD (
  CONSTRAINT NEWS_CONT_P_PK
  PRIMARY KEY
  (YMD, SEQNO, NEWSCODE, LINENO)
  USING INDEX PNWSOWN.NEWS_CONT_P_PK
  ENABLE VALIDATE);


CREATE OR REPLACE PUBLIC SYNONYM NEWS_CONT_P FOR PNWSOWN.NEWS_CONT_P;


GRANT SELECT ON PNWSOWN.NEWS_CONT_P TO RL_NWS_USR;
