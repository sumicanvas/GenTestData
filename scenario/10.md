## 시나리오 10번 : Highlight 검색 

  - 검색어와 일치하는 부분을 highlight 할 수 있도록 찾아주는 기능
    
```js
db.news_array.aggregate([
  {
    $search: {
      index: "news_search_index",
      compound: {
        must: [
          {
            text: {
              query: "삼성전자",
              path: ["title", "contents"],
              matchCriteria: "all"
            }
          }
        ]
      },
      highlight: {
        path: ["title", "contents"]
      },
      sort: {
        score: { $meta: "searchScore" },
        newscode_ts: -1
      }
    }
  },
  { $limit: 10 },
  {
    $project: {
      _id: 1,
      newscode_ts: 1,
      title: 1,
      contents: 1,
      dgubun: 1,
      shcode: 1,
      kind: 1,
      score: { $meta: "searchScore" },
      highlights: { $meta: "searchHighlights" }
    }
  }
]);

```
