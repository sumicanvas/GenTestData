## 시나리오 9번 : fuzzy 검색 

  - 한 글자를 잘못입력해도 제대로 찾아지도록 검색
  - 최종코드 : should 로  title과 contents 를 주고 ***boost 값으로 title에 가중치*** 부여
```js
db.news_array.aggregate([
  {
    $search: {
      index: "news_search_index",
      compound: {
        should: [
          {
            text: {
              query: "삼성전차",
              path: "title",
              matchCriteria: "all",
              fuzzy: {
                maxEdits: 1,
                prefixLength: 1,
                maxExpansions: 50
              },
              score: {
                boost: {
                  value: 5
                }
              }
            }
          },
          {
            text: {
              query: "삼성전차",
              path: "contents",
              matchCriteria: "all",
              fuzzy: {
                maxEdits: 1,
                prefixLength: 1,
                maxExpansions: 50
              },
              score: {
                boost: {
                  value: 1
                }
              }
            }
          }
        ],
        minimumShouldMatch: 1
      },
      sort: {
        score: { $meta: "searchScore" },
        newscode_ts: -1
      }
    }
  },
  {
    $addFields: {
      score: { $meta: "searchScore" }
    }
  },
  {
    $match: {
      score: { $gte: 1 }
    }
  },
  { $limit: 10 },
  {
    $project: {
      _id: 1,
      newscode_ts: 1,
      title: 1,
      contents: 1,
      dgubun: 1,
      shcode: 1,
      kind: 1,
      score: 1
    }
  }
]);

```
