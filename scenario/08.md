## 시나리오 8번 : 특정뉴스원 

  - 종목코드 X
  - 뉴스구분 O
  - 검색어 O
  - 특정 뉴스구분 전체 뉴스 최신순 조회
    
```js
db.news_array.aggregate([
  {
    $search: {
      index: "news_search_index",
      compound: {
        must: [
          {
            text: {
              query: "해운 업종",
              path: ["title", "contents"],
              matchCriteria: "all"
            }
          }
        ],
        filter: [
          {
            equals: {
              path: "dgubun",
              value: "M"
            }
          }
        ]
      },
      sort: {
        score: { $meta: "searchScore" },
        newscode_ts: -1
      }
    }
  },
  { $limit: 10 },
  {
    $project: {
      _id: 1,
      newscode_ts: 1,
      title: 1,
      contents: 1,
      dgubun: 1,
      shcode: 1,
      kind: 1,
      score: { $meta: "searchScore" }
    }
  }
]);

```
